    45→/** Render a title with the highlight word wrapped in .text-brand-gradient. */
    46→function HighlightTitle({ title, highlight }: { title: string; highlight?: string }) {
    47→  if (!highlight) return <>{title}</>;
    48→  const idx = title.toLowerCase().indexOf(highlight.toLowerCase());
    49→  if (idx === -1) return <>{title}</>;
    50→  const before = title.slice(0, idx);
    51→  const match = title.slice(idx, idx + highlight.length);
    52→  const after = title.slice(idx + highlight.length);
    53→  return (
    54→    <>
    55→      {before}
    56→      <span className="text-brand-gradient">{match}</span>
    57→      {after}
    58→    </>
    59→  );
    60→}
